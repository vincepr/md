        Preis = preis;
        Jahr = jahr;
    }
    public override string ToString(){
        return $"Titel: {Titel}\nPreis: {Preis}\nJahr: {Jahr}\nGenre: {Genre}\n";
    }
    public override bool Equals(object other){
        if (other is Buch){
            Buch o = (Buch)other;
            return Titel == o.Titel && Preis == o.Preis && Jahr == o.Jahr && Genre == o.Genre;
        }
        return base.Equals(other);
    }

    // IComparable<T> Interface erlaubt es direkt 2 EintrÃ¤ge zu vergleichen und damit direkt 'list.Sort()' aufzurufen.
    /// -1  -> other vor this
